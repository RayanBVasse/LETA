DROP TABLE IF EXISTS blogs_sample;

CREATE TABLE blogs_sample (
    post_id BIGINT AUTO_INCREMENT PRIMARY KEY,
    id BIGINT,
    gender VARCHAR(20),
    age INT,
    topic VARCHAR(200),
    sign VARCHAR(50),
    date_raw VARCHAR(50),
    text LONGTEXT CHARACTER SET utf8mb4 COLLATE utf8mb4_general_ci
) ENGINE=InnoDB
  DEFAULT CHARSET=utf8mb4
  COLLATE=utf8mb4_general_ci;

################################

LOAD DATA INFILE 
'C:/ProgramData/MySQL/MySQL Server 8.0/Uploads/blogs_clean_200k.csv'
INTO TABLE blogs_sample
CHARACTER SET utf8mb4
FIELDS TERMINATED BY ',' 
ENCLOSED BY '"'
ESCAPED BY '\\'
LINES TERMINATED BY '\n'
IGNORE 1 LINES
(id, gender, age, topic, sign, date_raw, text);


####################################

Check for any leftover “fake header” row
SELECT *
FROM blogs_sample
WHERE gender = 'gender'
   OR topic  = 'topic'
   OR sign   = 'sign'
   OR date_raw = 'date'
LIMIT 10;


If this returns 0 rows → perfect.

###################################

Posts per author (to see if we have good longitudinal arcs)
SELECT 
    id,
    COUNT(*) AS posts_per_author
FROM blogs_sample
GROUP BY id
ORDER BY posts_per_author DESC
LIMIT 20;


This gives you the top 20 “heaviest” authors — crucial for picking good case vignettes later.

#####################################
(d) Age distribution (to see who we’re actually modelling)
SELECT 
    age,
    COUNT(*) AS posts
FROM blogs_sample
GROUP BY age
ORDER BY age;


You’ll very likely see clusters around teens, 20s, 30s — that will matter later when we start comparing narrative styles by age.

######################################
(e) Topic distribution (how this slice is thematically shaped)
SELECT 
    topic,
    COUNT(*) AS posts
FROM blogs_sample
GROUP BY topic
ORDER BY posts DESC
LIMIT 20;

#####################################

SELECT 
    MAX(CHAR_LENGTH(text)) AS longest_text_mysql,
    AVG(CHAR_LENGTH(text)) AS avg_length_mysql
FROM blogs_sample;
This tells you:

the maximum length of any blog post stored in MySQL

the average length of blogs in the MySQL table
